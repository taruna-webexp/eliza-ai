pnpm self-update

# Install Dependencies
pnpm install
pnpm install --no-frozen-lockfile  # Use if lockfile causes issues

# Build the Project
pnpm build

# If build fails:

pnpm approve-builds  # Run this command
# Press 'a' and Enter, then press 'y' and Enter

# Start the Application
pnpm start

# Open a second terminal tab for the client
pnpm start:client
